# 신경망 모델의 검증
# predict(신경망 모델, 검증용 데이터 셋, type)
# 범주형 변수의 경우 class를 입력
predicted <- predict(model.nnet, data.test, type = "class")
# 검증용 데이터셋의 예측 결과가 벡터의 형태로 return
actural <- data.test$Species
# 오분류표 생성 (confusion matrix)
# table(실제 결과, 예측결과) 실제결과와 예측결과에 대한 교차표 생성
confusion.matrix <- table(actural, predicted)
confusion.matrix
# 모델의 정확도
paste("예측확률 :" ,round(mean(predicted == actural), 2) * 100,"%")
